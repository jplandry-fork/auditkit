Installation

# Quick install (Go required) go install github.com/guardian-nexus/auditkit/scanner/cmd/auditkit@latest
# Or build from source git clone https://github.com/guardian-nexus/auditkit.git cd auditkit/scanner && go build ./cmd/auditkit
# Run CMMC Level 1 assessment ./auditkit scan -framework cmmc -format pdf -output cmmc-evidence.pdf
# Multi-framework scan ./auditkit scan -framework all -format pdf -output compliance-report.pdf
# Azure SOC2 scan az login export AZURE_SUBSCRIPTION_ID="your-subscription-id" ./auditkit scan -provider azure -framework soc2
# GCP scan (NEW in v0.7.0) gcloud auth application-default login ./auditkit scan -provider gcp -framework soc2
